% 定义一个本地的空宏用于后续的条件比较 (判断 \rightmark 是否为空)
\def\myemptytestmacro{}

% 核心修改：重新定义 \subsectionmark 以更新 \rightmark
% 标准的 book.cls (ctexbook 基于此) 中 \subsectionmark 默认不做任何事
% 我们需要它像 \sectionmark 一样更新 \rightmark，以便页眉能显示小节标题
\makeatletter % 允许在宏定义中使用 @ 符号 (LaTeX内部命令常用)
\renewcommand{\subsectionmark}[1]{% 定义 \subsectionmark 命令，它接收一个参数 #1 (即小节标题)
  % \markright 命令用于设置 \rightmark 的内容
  % \ifnum\c@secnumdepth > \@ne ... \fi 判断小节是否编号 (secnumdepth > 1)
  % \thesubsection 是小节编号, \quad 是一个标准宽度的空格
  % #1 是小节标题本身
  \markright{\ifnum\c@secnumdepth > \@ne \thesubsection\quad\fi #1}%
}
\makeatother % 恢复 @ 符号的普通含义
% --- \subsectionmark 定义结束 ---

% 奇数页内侧页眉 (左上 LO = Left Odd)
\fancyhead[LO]{%
    \begin{varwidth}{\linewidth}% 使用 varwidth 使内容宽度自适应并能换行
    \raggedright% 左对齐
    \slshape\nouppercase{\leftmark}% 打印章标题 (\leftmark 通常由 \chaptermark 设置)
                                  % \nouppercase 防止 fancyhdr 默认将西文转为大写
    % 下面的逻辑判断是否需要在第二行打印 \rightmark (节或小节标题)
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}% 安全地获取 \leftmark 的内容用于比较
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}% 安全地获取 \rightmark 的内容
    \ifx\expandedrightmark\myemptytestmacro % 如果 \rightmark 为空 (例如某些特殊页面)
        % 则第二行不打印任何内容
    \else % \rightmark 不为空
        \ifx\expandedleftmark\expandedrightmark % 如果 \leftmark 和 \rightmark 内容相同
                                               % (例如，在章首页，\markboth 可能将两者设为相同)
            % 则第二行不打印，避免重复
        \else
            % \leftmark 和 \rightmark 内容不同，且 \rightmark 不为空，则换行打印 \rightmark
            \leavevmode\\ \small\slshape\nouppercase{\rightmark}% \leavevmode确保能正确换行，\small设置小号字体
        \fi
    \fi
    \end{varwidth}%
}

% 偶数页内侧页眉 (右上 RE = Right Even)
\fancyhead[RE]{%
    \begin{varwidth}{\linewidth}%
    \raggedleft% 右对齐
    \slshape\nouppercase{\leftmark}% 打印章标题
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%
    \ifx\expandedrightmark\myemptytestmacro
    \else
        \ifx\expandedleftmark\expandedrightmark
        \else
            \leavevmode\\ \small\slshape\nouppercase{\rightmark}%
        \fi
    \fi
    \end{varwidth}%
}

% 使章首页 (通常是 plain 页面样式) 也使用我们自定义的页眉页脚
% 这样章首页的页眉也会显示章标题和可能的节/小节标题 (如果紧随其后)

  % 页码设置同上
  \fancyhead[RO]{\slshape\thepage}%
  \fancyhead[LE]{\slshape\thepage}%
  % 内侧页眉逻辑同上
  \fancyhead[LO]{%
    \begin{varwidth}{\linewidth}
    \raggedright
    \slshape\nouppercase{\leftmark}%
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%
    \ifx\expandedrightmark\myemptytestmacro \else \ifx\expandedleftmark\expandedrightmark \else
    \leavevmode\\ \small\slshape\nouppercase{\rightmark}%
    \fi \fi
    \end{varwidth}%
  }%
  \fancyhead[RE]{%
    \begin{varwidth}{\linewidth}
    \raggedleft
    \slshape\nouppercase{\leftmark}%
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%
    \ifx\expandedrightmark\myemptytestmacro \else \ifx\expandedleftmark\expandedrightmark \else
    \leavevmode\\ \small\slshape\nouppercase{\rightmark}%
    \fi \fi
    \end{varwidth}%
  }%
  \renewcommand{\headrulewidth}{0pt}% 章首页不显示页眉顶部的分隔线
  \renewcommand{\footrulewidth}{0pt}% 章首页不显示页脚底部的分隔线 (如果fancyhf清空了页脚则此行意义不大)
