% --- 页眉页脚设置 ---
\pagestyle{fancy}

% 外侧页眉：页码
\fancyhead[RO]{\slshape\thepage} % 奇数页，右边（外侧）
\fancyhead[LE]{\slshape\thepage} % 偶数页，左边（外侧）

% 定义一个本地的空宏用于比较
\def\myemptytestmacro{}

% 奇数页内侧页眉 (左上 LO)
\fancyhead[LO]{%
    \begin{varwidth}{\linewidth}
    \raggedright
    \slshape\nouppercase{\leftmark}% 打印章标题 (Chapter)

    % 比较 \leftmark 和 \rightmark 的实际内容 (转换成字符串后比较)
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%

    \ifx\expandedrightmark\myemptytestmacro % 如果 \rightmark 为空 (没有内容)
        % 不打印 \rightmark
    \else % \rightmark 不为空
        \ifx\expandedleftmark\expandedrightmark % 如果 \leftmark 和 \rightmark 内容相同 (通常是章首页且无节)
            % 不打印 \rightmark，避免重复
        \else
            % \leftmark 和 \rightmark 内容不同，且 \rightmark 不为空，则换行打印 \rightmark (Section)
            \leavevmode\\ \small\slshape\nouppercase{\rightmark}% 使用 \leavevmode\\ 确保换行安全
        \fi
    \fi
    \end{varwidth}%
}

% 偶数页内侧页眉 (右上 RE)
\fancyhead[RE]{%
    \begin{varwidth}{\linewidth}
    \raggedleft % 右对齐
    \slshape\nouppercase{\leftmark}% 打印章标题 (Chapter)

    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%

    \ifx\expandedrightmark\myemptytestmacro
    \else
        \ifx\expandedleftmark\expandedrightmark
        \else
            \leavevmode\\ \small\slshape\nouppercase{\rightmark}% 打印节标题 (Section)
        \fi
    \fi
    \end{varwidth}%
}

% 使章首页 (plain 样式)也使用自定义页眉，但不显示分割线
\fancypagestyle{plain}{%
  \fancyhf{}% 清空所有页眉页脚
  \fancyhead[RO]{\slshape\thepage}% 奇数页，外侧页码
  \fancyhead[LE]{\slshape\thepage}% 偶数页，外侧页码
  \fancyhead[LO]{% 奇数页，内侧内容
    \begin{varwidth}{\linewidth}
    \raggedright
    \slshape\nouppercase{\leftmark}%
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%
    \ifx\expandedrightmark\myemptytestmacro \else \ifx\expandedleftmark\expandedrightmark \else
    \leavevmode\\ \small\slshape\nouppercase{\rightmark}%
    \fi \fi
    \end{varwidth}%
  }%
  \fancyhead[RE]{% 偶数页，内侧内容
    \begin{varwidth}{\linewidth}
    \raggedleft
    \slshape\nouppercase{\leftmark}%
    \edef\expandedleftmark{\detokenize\expandafter{\leftmark}}%
    \edef\expandedrightmark{\detokenize\expandafter{\rightmark}}%
    \ifx\expandedrightmark\myemptytestmacro \else \ifx\expandedleftmark\expandedrightmark \else
    \leavevmode\\ \small\slshape\nouppercase{\rightmark}%
    \fi \fi
    \end{varwidth}%
  }%
  \renewcommand{\headrulewidth}{0pt}% 无页眉分割线
  \renewcommand{\footrulewidth}{0pt}% 无页脚分割线
}
